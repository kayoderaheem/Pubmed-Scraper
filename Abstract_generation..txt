from Bio import Entrez

# Set your email address for Entrez
Entrez.email = "raheemkayode2017@gmail.com"

# Define the molecular target
molecular_target = "EGFR, P53"

# Search for information on the molecular target in PubMed
def search_pubmed(query):
    handle = Entrez.esearch(db="pubmed", term=query)
    record = Entrez.read(handle)
    handle.close()
    return record

# Retrieve abstracts of publications related to the molecular target
def get_abstracts(pubmed_ids):
    handle = Entrez.efetch(db="pubmed", id=pubmed_ids, rettype="abstract", retmode="text")
    abstracts = handle.read()
    handle.close()
    return abstracts

# Search for molecular target-related publications
search_query = f"{molecular_target} AND cancer"
search_results = search_pubmed(search_query)

# Retrieve abstracts for the first five publications
pubmed_ids = search_results["IdList"][:5]
abstracts = get_abstracts(pubmed_ids)

# Print the abstracts
for i, abstract in enumerate(abstracts.split("\n\n")):
    print(f"Abstract {i+1}:")
    print(abstract)
    print("--------------------------------------------------")
